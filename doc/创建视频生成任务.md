# 创建视频生成任务

## OpenAPI Specification

```yaml
openapi: 3.0.1
info:
  title: ''
  description: ''
  version: 1.0.0
paths:
  /v1/videos/generations:
    post:
      summary: 创建视频生成任务
      deprecated: false
      description: ''
      tags:
        - API文档/SORA
      parameters:
        - name: Authorization
          in: header
          description: ''
          required: false
          example: your_api_key_here
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                model:
                  type: string
                  title: 模型
                  enum:
                    - sora-2
                    - sora-2-pro
                    - sora-2-temporary
                  x-apifox-enum:
                    - value: sora-2
                      name: 常用版本
                      description: 目前不可用
                    - value: sora-2-pro
                      name: pro版
                      description: ''
                    - value: sora-2-temporary
                      name: sora-2常用版本的临时通道,支持10秒/15秒
                      description: 价格很贵,请急用的时候再接入,不支持25秒,25秒请用sora-2-pro,pro是一直基本可用的状态
                prompt:
                  type: string
                  title: 提示词
                aspect_ratio:
                  type: string
                  title: 视频宽高比
                  description: 16:9或9:16
                duration:
                  type: integer
                  title: 时长
                  enum:
                    - 10
                    - 15
                    - 25
                  x-apifox-enum:
                    - value: 10
                      name: ''
                      description: sora-2是标清,当sora-2-pro时为高清
                    - value: 15
                      name: ''
                      description: sora-2是标清,当sora-2-pro时为高清
                    - value: 25
                      name: 仅sora-2-pro支持,25秒是标清
                      description: ''
                  default: 10
                image_urls:
                  type: array
                  items:
                    type: string
                    description: 要公网能访问的图片链接
                  title: 参考图
                  description: 选填,数组形式,目前仅支持1张,传法请看示例
                character_url:
                  type: string
                  title: 创建角色需要的视频链接
                  description: 公网,可访问,可下载,需要有音频通道,不能出现真人
                  deprecated: true
                character_timestamps:
                  type: string
                  title: 视频角色出现的秒数范围
                  description: 以,分割,比如0至3秒设置为0,3
                  deprecated: true
                callback_url:
                  type: string
                  title: 回调地址
                characters:
                  type: string
                  title: 多角色客串
                  examples:
                    - '[{ url: "https://xxx.com/xxx.mp4", timestamps: "0,3" }]'
                  description: |-

                    该功能不支持传真人视频
                    参数url: 角色视频url(要互联网访问并可下载)
                    参数timestamps:以,英文逗号分割,比如0至3秒设置为0,3,最长支持3秒
                    传多个角色,默认全部引用.同时也支持指定某个角色引用
                    格式:
                    @character1 ，数字1指角色在数组的位置第一个
                    从1开始,不是索引0开始
                    数组长度最多3个
                    (建议@角色后留个空格)
              required:
                - model
                - prompt
              x-apifox-orders:
                - model
                - prompt
                - aspect_ratio
                - duration
                - image_urls
                - characters
                - callback_url
                - character_url
                - character_timestamps
            examples:
              '1':
                value:
                  model: sora-2
                  prompt: The car moves forward at a high speed
                  aspect_ratio: '16:9'
                  duration: 15
                  image_urls:
                    - >-
                      https://crea-img.xinzhiyun.net/upload/202502251100509684.jpg
                summary: 视频生成
              '2':
                value:
                  model: sora-2
                  prompt: The car moves forward at a high speed
                  aspect_ratio: '16:9'
                  duration: 15
                  image_urls:
                    - >-
                      https://crea-img.xinzhiyun.net/upload/202502251100509684.jpg
                  characters:
                    - url: >-
                        https://v3.fal.media/files/penguin/hsR_KXBJjuF3IIVYIIDA2_output.mp4
                      timestamps: 0,3
                    - url: >-
                        https://v3.fal.media/files/penguin/hsR_KXBJjuF3IIVYIIDA2_output.mp4
                      timestamps: 0,3
                  callback_url: https://webhook.site/f95d67b6-cbc4-475d-9ea8-93e3c1786d45
                summary: 多角色客串
              '3':
                value:
                  model: sora-2
                  prompt: 唱歌
                  aspect_ratio: '16:9'
                  image_urls:
                    - >-
                      https://crea-img.xinzhiyun.net/upload/202502251100509684.jpg
                  duration: 15
                  callback_url: https://webhook.site/2ef1a017-6ebd-4f15-9c9a-5ac3652e4abd
                summary: 图生视频
      responses:
        '200':
          description: ''
          content:
            application/json:
              schema:
                type: object
                properties:
                  id:
                    type: string
                required:
                  - id
                x-apifox-orders:
                  - id
          headers: {}
          x-apifox-name: 成功
      security: []
      x-apifox-folder: API文档/SORA
      x-apifox-status: released
      x-run-in-apifox: https://app.apifox.com/web/project/4804237/apis/api-357826026-run
components:
  schemas: {}
  securitySchemes: {}
servers:
  - url: https://duomiapi.com
    description: 开发环境
security: []

```
